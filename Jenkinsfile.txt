pipeline{

	agent any
		
		stages {
		
			stage('One') {
			
						steps {
								echo 'Hello Pipeline'
						}
			}
			stage('Two') {
			
						steps {
								input('Do you want to proceed')
						}
			}
			stage('Three') {
						
						when{
							not{
								branch 'phase-2'
							}
						}
						steps{
							echo "Hello stage three"
						}
			}
			stage('Four') {
						  parallel{
								stage('Unit Test'){
										agent{
											steps{
											echo "Unit test step"
											}
										}
								}
								stage('Integration Test'){
										steps{
											echo "Running Integration test"
										}	
								}
						  }
			}
		}

}
