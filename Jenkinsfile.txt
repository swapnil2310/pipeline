pipeline{

	agent any
		
		stages {
		
			stage('One') {
			
						steps {
								echo 'Hello Pipeline'
						}
			}
			stage('Two') {
			
						steps {
								input('Do you want to proceed')
						}
			}
			stage('Three') {
						
						when{
							not{
								branch 'phase-2'
							}
						}
						steps{
							echo "Hello stage three"
						}
			}
			
		}

}
